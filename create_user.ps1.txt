param (
    [int]$count = 20
)

Import-Module ActiveDirectory

$usersData = @(
    @{ prefix = "mkt";   ou = "OU=MKT,DC=alma,DC=local";   group = "MKT";   pass = "Skill39!" }
    @{ prefix = "sales";  ou = "OU=SALES,DC=alma,DC=local"; group = "SALES"; pass = "Skill39!" }
    @{ prefix = "tech";   ou = "OU=TECH,DC=alma,DC=local";  group = "TECH";  pass = "Skill39!" }
    @{ prefix = "hr";     ou = "OU=HR,DC=alma,DC=local";    group = "HR";    pass = "Skill39!" }
)

foreach ($data in $usersData) {
    for ($i = 1; $i -le $count; $i++) {
        $username = "$($data.prefix)$i"

        if (Get-ADUser -Filter "SamAccountName -eq '$username'") {
            continue
        }

        $securePassword = ConvertTo-SecureString $data.pass -AsPlainText -Force

        New-ADUser -Name $username `
                   -SamAccountName $username `
                   -UserPrincipalName "$username@alma.local" `
                   -Path $data.ou `
                   -AccountPassword $securePassword `
                   -ChangePasswordAtLogon $false `
                   -Enabled $true

        Add-ADGroupMember -Identity $data.group -Members $username
    }
}
